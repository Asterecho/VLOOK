    /**
     * 创建背景水印
     */
    function createWatermark() {
        var fontSize = 28;
        var fontSetting = fontSize + "px Microsoft YaHei";
        // temp canvas
        $("body").append('<canvas id="tempCanvas" style="display: none"></canvas>');
        var tmpCanvas = document.getElementById("tempCanvas");
        var tmpCtx = tmpCanvas.getContext("2d");
        tmpCtx.font = fontSetting;
        // canvas
        var a = -30;
        var r = a * Math.PI / 180;
        var z = tmpCtx.measureText(document.title).width;
        var w = z;
        var h = Math.abs(z * Math.sin(180 - a)) + fontSize * 1.5;
        // console.log(w + "," + h);
        $("body").append('<canvas id="CanvasMask" width="' + w + '" height="' + h + '" style="display: none"></canvas>');
        var canvas = document.getElementById("CanvasMask");
        var ctx = canvas.getContext("2d");
        ctx.font = fontSetting;
        // ctx.strokeStyle = "#FF0000";
        // ctx.strokeRect(0, 0, w, h);
        ctx.translate(0, h);
        ctx.rotate(r);
        // 根据背景色亮度设置水印文字颜色
        // console.log($("body").css("background-color").replace("rgb(", "").replace(")", "").split(","));
        var bgColor = $("body").css("background-color").replace("rgb(", "").replace(")", "").split(",");
        lightValue = parseInt(bgColor[0]) + parseInt(bgColor[1]) + parseInt(bgColor[2]);
        if (lightValue < 384) // 暗色背景
            ctx.fillStyle = "rgba(255, 255, 255, 0.05)";
        else
            ctx.fillStyle = "rgba(0, 0, 0, 0.05)";
        ctx.fillText(document.title, fontSize / 2, 0);
        ctx.restore();
        var img = new Image();
        img.src = canvas.toDataURL("image/png");
        $("body").css("background", "url(" + img.src + ")");
        $("body").css("background-repeat", "repeat");
    }

    /**
     * 页面滚动时高亮outline中的对应章节项
     */
    function followHeader() {
        var sTop = $("#toc-box-body").scrollTop();
        var sBottom = sTop + $("#toc-box-body").height();
        var currentIndex = -1;

        // 寻找当前屏幕显示的章节
        for (var i = 0; i < _headers.length; i++) {
            var obj = $("a[name='" + _headers[i] + "']");
            var headerHeight = parseInt(obj.parent().css("height"));
            /* console.log($("#oli-" + _headers[i]).text() + ": " + $(document).scrollTop() + ", " + hotArea); */
            if ((obj.offset().top - $(document).scrollTop()) > (headerHeight / 2)) { // 当前章节标题已滚过了屏幕顶部
                currentIndex = i;
                break;
            }
        }

        // 当前章节未变更
        if (_lastCurrentIndex == currentIndex)
            return;

        _lastCurrentIndex = currentIndex;
        // 当前屏幕内容为第1章或以后章节的内容
        if (currentIndex > 0 || // 非最后章节
            (currentIndex == -1 && _headers.length > 1)) { // 最后章节
            // 根据是否为最后章节修正当前章节索引
            currentIndex = currentIndex > 0 ? (currentIndex - 1) : (_headers.length - 1);

            if (_lastItem != null) // 清除大纲中上一次当前章节的样式
                _lastItem.removeClass("mdx-header-current");

            _lastItem = $("#oli-" + _headers[currentIndex]);
            _lastItem.addClass("mdx-header-current");
            // expandBubble(_lastItem.attr("id")); // 如果节点没有展开，则自动展开
            var preDis = _lastItem.height() * 2;

            if (_lastItem.position().top < sTop)
                $("#toc-box-body").scrollTop(_lastItem.position().top);
            else if (_lastItem.position().top > (sBottom - preDis))
                $("#toc-box-body").scrollTop(_lastItem.position().top - $("#toc-box-body").height() + preDis);
            // 显示面包屑导航条
            $(".mdx-bread-nav-text").text(_lastItem.text());
            // console.log(_lastItem.text());
            $(".mdx-bread-nav").fadeIn();
            $(".mdx-docbar-top").css("top", $(".mdx-bread-nav").css("height"));
            $(".mdx-figure-launcher").animate({
                top: "60px"
            }, _aniSpeed);

            if (currentIndex > 0) {
                $(".mdx-nav-prev").css("display", "block");
                $(".mdx-nav-prev-text").text($("#oli-" + _headers[currentIndex - 1]).text());
                $(".mdx-nav-prev-text").attr("data-anchor", _headers[currentIndex - 1]);
            } else
                $(".mdx-nav-prev").css("display", "none");

            if (currentIndex < _headers.length - 1) {
                $(".mdx-nav-next").css("display", "block");
                $(".mdx-nav-next-text").text($("#oli-" + _headers[currentIndex + 1]).text());
                $(".mdx-nav-next-text").attr("data-anchor", _headers[currentIndex + 1]);
            } else
                $(".mdx-nav-next").css("display", "none");
        }
        else { // 隐藏面包屑导航条
            $(".mdx-bread-nav").fadeOut();
            $(".mdx-docbar-top").css("top", "0px");
            $(".mdx-figure-launcher").animate({
                top: "10px"
            }, _aniSpeed);
        }
    }

    /**
     * 收起/展开章节目录
     * @param id 对象的id值
     */
    function folding(id) {
        var obj = $("#" + id);
        var folder = $("#fd-" + id);
        var folding = obj.attr("data-folding");
        if (folding == "0") {
            obj.attr("data-folding", "1");
            folder.html(_unfoldChar);
            disposeFolding(id, 1, true);
        } else if (folding == "1") {
            obj.attr("data-folding", "0");
            folder.html(_foldChar);
            disposeFolding(id, 0, true);
        }
    }
    /**
     * 处理展开或收起指定ID下的子章节
     * @param id 对象的id值
     * @param act 动作：0=收起，1=展开
     * @param recu 是否遍历子元素
     */
    function disposeFolding(id, act, recu) {
        var lastItem = null;
        var itemSet = $("#" + id).nextAll();
        for (var i = 0; i < itemSet.length; i++) {
            var item = $(itemSet[i]);
            if (lastItem != null) {
                var hd1 = getCSSHeaderNum(item);
                var hd2 = getCSSHeaderNum(lastItem);
                if (hd1 > hd2) continue;
                else if (hd1 < hd2) break;
            }
            // 如果是目录节点，同时是已展开的，才进行递归调用展开子节点
            if (recu == true && item.attr("data-node") == "1" && item.attr("data-folding") == "1") {
                disposeFolding(item.attr("id"), act, recu);
            }
            if (act == 0) // 收起
                item.css("display", "none");
            else if (act == 1) // 展开
                item.css("display", "block");
            lastItem = item;
        }
    }
    /**
     * 返回指定对象的CSS的header的样式层级

    function getCSSHeaderNum(obj) {
        var cname = obj.attr("class");
        return cname.substr(cname.indexOf("md-toc-h") + "md-toc-h".length, 1)
    }

            }
            if (item.attr("data-node") == "1" && item.attr("data-folding") == "1") {
                $("#fd-" + item.attr("id")).html(_unfoldChar);
                disposeFolding(item.attr("id"), 1, true);
            }
            item.css("display", "block");
            lastItem = item;
        }
    }
    /**
     * 更新指定内容的code标记样式
     */
    function renderingCodeStyle() {
        // 更新指定内容的<code>标签样式
        $("code").each(function() {
            if ($(this).text().length <= 3)
                return;

            // 将全角竖线替换为半角
            $(this).text($(this).text().replaceAll("｜", "|"));
            var codeText = $(this).text();

            // 分类式样式（浅色）
            var codeTags = new Array("==|", "++|", "**|", "--|", "^^|", "##|");
            var flag = false;
            for (var i=0; i<codeTags.length; i++) {
                if (codeText.indexOf(codeTags[i]) == 0) {
                    rebuildSpecCode($(this), codeTags[i], i + 1, false);
                    flag = true;
                    break;
                }
            }
            // 分类式样式（深色）
            if (flag == true) return;
            codeTags = new Array("=|", "+|", "*|", "-|", "^|", "#|");
            flag = false;
            for (var i=0; i<codeTags.length; i++) {
                if (codeText.indexOf(codeTags[i]) == 0) {
                    rebuildSpecCode($(this), codeTags[i], i + 1, true);
                    flag = true;
                    break;
                }
            }
            // 分段式样式
            if (flag == true) return;
            codeTags = new Array("|", "||", "|||", "||||", "|||||", "||||||");
            for (var i=5; i>=0; i--) {
                if (codeText.indexOf(codeTags[i]) > 0) {
                    rebuildSegmentCode($(this), codeTags[i], i+1);
                    break;
                }
            }
        });
    }
    /**
     * 替换所有内容
     */
    String.prototype.replaceAll = function(f,e){
        var reg = new RegExp(f, "g");
        return this.replace(reg, e);
    }
    /**
     * 以分类式重新设置code的渲染样式
     */
    function rebuildSpecCode(obj, split, number, flag) {
        var codeText = obj.text();
        var si = codeText.indexOf(split);
        var alt = flag ? "-alt" : "";
        obj.text(codeText.substr(si + split.length, codeText.length));
        obj.attr("class", "mdx-code-c" + number + alt);
    }
    /**
     * 以分段式重建code标签结构与内容
     */
    function rebuildSegmentCode(obj, split, number) {
        var codeText = obj.text();
        var si = codeText.indexOf(split);
        obj.before("<code class='mdx-code-name" + number + "'>" + codeText.substr(0, si) + "</code>");
        obj.text(codeText.substr(si + split.length, codeText.length));
        obj.attr("class", "mdx-code-value" + number);
    }
    /**
     * 更新指定内容的mermaid时序图的角色样式
     */
    function updateMermaidStyle() {

        $(".md-diagram-panel > svg[id^='mermaidChart'] > g > rect[class='actor']").each(function() {
            var textObj = $(this).next().children(); // actor rect object
            var person = /\@.*/g;
            var bgSys = /\/.*\//g;
            var extSys = /\_.*\_/g;

            // 更新<人物角色>样式
            if (person.test(textObj.text()) == true) {
                $(this).css("fill", $("body").css("background-color")); // actor / rect
                // $(this).css("stroke", $("body").css("background-color")); // actor / rect
                textObj.css("fill", $(".mdx-master-color").css("color")); // actor text
                var h = parseInt($(this).css("height"));
                $(this).css("rx", h / 2);
                $(this).css("ry", h / 2);
            }
            // 更新 /后端支撑系统/ 样式
            else if (bgSys.test(textObj.text()) == true) {
                $(this).css("stroke-width", "2"); // actor / rect
                textObj.css("font-weight", "bold"); // actor text
                // $(this).css("fill", $("body").css("color")); // actor / rect
                // textObj.css("fill", $("body").css("background-color")); // actor text
            }
            // 更新 _外部系统_ 样式
            else if (extSys.test(textObj.text()) == true) {
                // $(this).css("fill", $("body").css("background-color")); // actor / rect
                // $(this).css("stroke-width", "2"); // actor / rect
                $(this).css("stroke-dasharray", "5 2"); // actor / rect
            }
        });

        // 更新loop/opt标签样式
        $(".md-diagram-panel > svg > g > text[class='labelText'] > tspan").each(function() {
            var textObj = $(this);
            var bgColor = $(".mdx-third-color").css("background-color");
            var borderColor = $(".mdx-third-color").css("border-color");
            var textColor = $(".mdx-third-color").css("color");
            if (textObj.text() == "loop") { // loop
                bgColor = $(".mdx-fourth-color").css("background-color");
                borderColor = $(".mdx-fourth-color").css("border-color");
                textColor = $(".mdx-fourth-color").css("color");
            }
            if (textObj.text() == "loop" || textObj.text() == "opt") {
                var g = textObj.parent().parent();
                g.children("polygon.labelBox").css("fill", bgColor);
                g.children("line.loopLine").css("stroke", borderColor);
                g.children("text.labelText").css("fill", textColor);
                g.children("text.labelText").children("tspan").css("fill", textColor);
                g.children("text.loopText").children("tspan").css("fill", textColor);
            }
        });
    }
